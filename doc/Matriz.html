def *(mat)
  if (@mCol == mat.nFil)
    result = Array.new
    (@nFil).times do |i|
      result[i] = Array.new
      (mat.mCol).times do |j|
        if(@matriz[i][j].class==Fraccion)
         result[i][j] = Fraccion.new(0,1)
        else
          result[i][j] = 0
        end
      end
    end

    aux = Matriz.new(result)

    (@nFil).times do |i|
      (mat.mCol).times do |j|
        (@mCol).times do |z|
          aux.matriz[i][j] += @matriz[i][z] * mat.matriz[z][j]
        end
      end
    end
  else
    aux = 0
  end
  aux
end

def +(mat)
  if (mat.nFil == @nFil && mat.mCol == @mCol)
    aux = Matriz.new(@matriz)
    (@nFil).times do |i|
      (@mCol).times do |j|
        aux.matriz[i][j] = @matriz[i][j] + mat.matriz[i][j]
      end
    end
  else
    aux = 0
  end
  aux
end

def det
  if (@nFil == @mCol)
  det = @matriz[0][0]
  aux = Matriz.new(@matriz)
  for k in 0...@nFil do
    l = k+1
    for i in l...@mCol do
      for j in l...@mCol do
        aux.matriz[i][j] = (aux.matriz[k][k] * aux.matriz[i][j] - aux.matriz[k][j] * aux.matriz[i][k])/ aux.matriz[k][k]
      end
    end
    det = det * aux.matriz[k][k]
  end
  else
  det = "ERROR, la matriz no es cuadrada"
  end
  det
end

def to_s
  matString = ""
  for i in 0...@nFil do
    for j in 0...@mCol do
      matString = matString + @matriz[i][j].to_s + " "
    end
    matString = matString + "\n"
  end
  matString
end

def tras
  result = Array.new
    for i in 0...@nFil do
      result[i] = Array.new
      for j in 0...@mCol do
        result[i][j] = 0
      end
    end

  aux = Matriz.new(result)
  for i in 0...@nFil do
    for j in 0...@mCol do
      aux.matriz[i][j] = @matriz[j][i]
    end
  end
  aux
end
