def % (other)
  x=(@n*other.d)
  y=(@d*other.n)
  max_ = gcd(x,y)

  y=y/max_

  else

def <=>(other)
  if ((@n == other.n) && (@d == other.d))
    return 0
  end
  if ((@n * other.d) < (@d * other.n))
    return -1
  end
  if ((@n * other.d) > (@d * other.n))
    return 1 
  end
end
